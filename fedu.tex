\phantomsection\label{ley-de-oferta}
\begin{block}{Teorema (Ley de Oferta)}
Para todo par de precios \( p, p' \geq 0 \), tal que \( p \neq p' \), la oferta de la firma verifica:
\[
(p - p') [ y(p) - y(p')] \geq 0
\]
\end{block}
\end{frame}
